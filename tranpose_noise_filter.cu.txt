#include<stdio.h>
#include<stdlib.h>
#include<cuda.h>

#define N 2048
#define BLOCK_SIZE 32
#define THRESHOLD 10

__global__ void lidar_transpose_filter(float *input, float *output)
{
    __shared__ float tile[BLOCK_SIZE][BLOCK_SIZE];

    int x = threadIdx.x + blockIdx.x * blockDim.x;
    int y = threadIdx.y + blockIdx.y * blockDim.y;

    int localX = threadIdx.x;
    int localY = threadIdx.y;

    if(x < N && y < N)
    {
        int index = y * N + x;

        float value = input[index];

        if(value < THRESHOLD)
            value = 0;

        tile[localX][localY] = value;
    }

    __syncthreads();

    int tx = threadIdx.x + blockIdx.y * blockDim.x;
    int ty = threadIdx.y + blockIdx.x * blockDim.y;

    if(tx < N && ty < N)
    {
        int tindex = ty * N + tx;
        output[tindex] = tile[localY][localX];
    }
}

void fill_data(float *data)
{
    for(int i=0;i<N*N;i++)
        data[i] = (float)(i%20);
}

int main()
{
    float *a,*b;
    float *d_a,*d_b;

    int size = N*N*sizeof(float);

    a = (float*)malloc(size);
    b = (float*)malloc(size);

    fill_data(a);

    cudaMalloc((void**)&d_a,size);
    cudaMalloc((void**)&d_b,size);

    cudaMemcpy(d_a,a,size,cudaMemcpyHostToDevice);

    dim3 block(BLOCK_SIZE,BLOCK_SIZE);
    dim3 grid(N/BLOCK_SIZE,N/BLOCK_SIZE);

    lidar_transpose_filter<<<grid,block>>>(d_a,d_b);

    cudaMemcpy(b,d_b,size,cudaMemcpyDeviceToHost);

    printf("LiDAR grid transpose with noise filtering done\n");

    cudaFree(d_a);
    cudaFree(d_b);

    free(a);
    free(b);

    return 0;
}
