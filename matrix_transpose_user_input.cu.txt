#include<stdio.h>
#include<stdlib.h>
#include<cuda.h>

#define BLOCK_SIZE 32

__global__ void matrix_transpose_shared(int *input, int *output, int N)
{
    __shared__ int tile[BLOCK_SIZE][BLOCK_SIZE];

    int x = threadIdx.x + blockIdx.x * blockDim.x;
    int y = threadIdx.y + blockIdx.y * blockDim.y;

    int localX = threadIdx.x;
    int localY = threadIdx.y;

    if(x < N && y < N)
    {
        int index = y * N + x;
        tile[localX][localY] = input[index];
    }

    __syncthreads();

    int tx = threadIdx.x + blockIdx.y * blockDim.x;
    int ty = threadIdx.y + blockIdx.x * blockDim.y;

    if(tx < N && ty < N)
    {
        int tindex = ty * N + tx;
        output[tindex] = tile[localY][localX];
    }
}

void fill_array(int *data, int N)
{
    for(int i = 0; i < N*N; i++)
        data[i] = i;
}

void print_matrix(int *data, int N)
{
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            printf("%4d ", data[i*N+j]);
        }
        printf("\n");
    }
}

int main()
{
    int N;

    printf("Enter matrix size N: ");
    scanf("%d",&N);

    int size = N*N*sizeof(int);

    int *a,*b;
    int *d_a,*d_b;

    a = (int*)malloc(size);
    b = (int*)malloc(size);

    fill_array(a,N);

    cudaMalloc((void**)&d_a,size);
    cudaMalloc((void**)&d_b,size);

    cudaMemcpy(d_a,a,size,cudaMemcpyHostToDevice);

    dim3 block(BLOCK_SIZE,BLOCK_SIZE);
    dim3 grid((N+BLOCK_SIZE-1)/BLOCK_SIZE,(N+BLOCK_SIZE-1)/BLOCK_SIZE);

    matrix_transpose_shared<<<grid,block>>>(d_a,d_b,N);

    cudaMemcpy(b,d_b,size,cudaMemcpyDeviceToHost);

    printf("\nOriginal Matrix:\n");
    print_matrix(a,N);

    printf("\nTransposed Matrix:\n");
    print_matrix(b,N);

    cudaFree(d_a);
    cudaFree(d_b);

    free(a);
    free(b);

    return 0;
}
