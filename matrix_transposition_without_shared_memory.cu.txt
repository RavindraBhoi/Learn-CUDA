#include<stdio.h>
#include<stdlib.h>
#include<cuda.h>

#define N 2048
#define BLOCK_SIZE 32

__global__ void radar_transpose(float *input,float *output)
{
    int x = threadIdx.x + blockIdx.x * blockDim.x;
    int y = threadIdx.y + blockIdx.y * blockDim.y;

    if(x < N && y < N)
    {
        int index = y*N + x;
        int tindex = x*N + y;

        output[tindex] = input[index];
    }
}

void fill_data(float *data)
{
    for(int i=0;i<N*N;i++)
        data[i] = (float)i;
}

int main()
{
    float *a,*b;
    float *d_a,*d_b;

    int size = N*N*sizeof(float);

    a = (float*)malloc(size);
    b = (float*)malloc(size);

    fill_data(a);

    cudaMalloc((void**)&d_a,size);
    cudaMalloc((void**)&d_b,size);

    cudaMemcpy(d_a,a,size,cudaMemcpyHostToDevice);

    dim3 block(BLOCK_SIZE,BLOCK_SIZE);
    dim3 grid(N/BLOCK_SIZE,N/BLOCK_SIZE);

    radar_transpose<<<grid,block>>>(d_a,d_b);

    cudaMemcpy(b,d_b,size,cudaMemcpyDeviceToHost);

    printf("Radar matrix transpose completed\n");

    cudaFree(d_a);
    cudaFree(d_b);

    free(a);
    free(b);

    return 0;
}
